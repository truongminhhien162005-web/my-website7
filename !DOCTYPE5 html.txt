<script>

const plane =
document.getElementById("plane");

const enemyPlane =
document.getElementById("enemyPlane");

const building =
document.getElementById("building");

const scoreText =
document.getElementById("score");

const startScreen =
document.getElementById("startScreen");

const gameOver =
document.getElementById("gameOver");

const finalScore =
document.getElementById("finalScore");

const replayBtn =
document.getElementById("replayBtn");

let started = false;
let running = true;

let planeY = 220;
let velocity = 0;

let buildingX =
window.innerWidth;

let buildingHeight = 180;

let enemyX =
window.innerWidth + 400;

let enemyY = 250;

let score = 0;
let speed = 6;

function createBuilding(){

    buildingHeight =
    Math.floor(Math.random()*120)
    + 120;

    building.style.height =
    buildingHeight + "px";

    buildingX =
    window.innerWidth;

    building.innerHTML = "";

    for(let row=0; row<8; row++){

        for(let col=0; col<3; col++){

            const w =
            document.createElement("div");

            w.className = "window";

            w.style.left =
            (15 + col*25) + "px";

            w.style.top =
            (10 + row*18) + "px";

            building.appendChild(w);
        }
    }
}

createBuilding();

document.addEventListener(
"keydown",
function(e){

    if(e.code !== "Space")
        return;

    e.preventDefault();

    if(!started){

        started = true;

        startScreen.style.display =
        "none";

        requestAnimationFrame(
            gameLoop
        );
    }

    if(!running)
        return;

    velocity = 13;
});

function endGame(){

    running = false;

    finalScore.innerHTML =
    "Điểm của bạn: "
    + score;

    gameOver.style.display =
    "flex";
}

function resetGame(){

    running = true;
    started = true;

    planeY = 220;
    velocity = 0;

    score = 0;
    speed = 6;

    scoreText.innerHTML = "0";

    enemyX =
    window.innerWidth + 400;

    enemyY = 250;

    createBuilding();

    gameOver.style.display =
    "none";

    requestAnimationFrame(
        gameLoop
    );
}

replayBtn.addEventListener(
"click",
resetGame
);

function gameLoop(){

    if(!running)
        return;

    velocity -= 0.55;

    planeY += velocity;

    if(planeY < 0){

        planeY = 0;
        velocity = 0;
    }

    plane.style.bottom =
    planeY + "px";

    if(velocity > 0){

        plane.style.transform =
        "rotate(-4deg)";
    }
    else{

        plane.style.transform =
        "rotate(4deg)";
    }

    /* Va chạm vùng cấm bay */

    if(
        planeY >
        window.innerHeight - 180
    ){
        endGame();
        return;
    }

    buildingX -= speed;

    building.style.left =
    buildingX + "px";

    if(buildingX < -120){

        score++;

        scoreText.innerHTML =
        score;

        speed += 0.12;

        createBuilding();
    }

    if(
        buildingX < 170 &&
        buildingX > 30 &&
        planeY <
        buildingHeight + 35
    ){
        endGame();
        return;
    }

    enemyX -= speed + 2;

    enemyPlane.style.left =
    enemyX + "px";

    enemyPlane.style.bottom =
    enemyY + "px";

    if(enemyX < -150){

        enemyX =
        window.innerWidth + 300;

        enemyY =
        Math.floor(Math.random()*250)
        + 80;
    }

    if(
        enemyX < 170 &&
        enemyX > 50 &&
        Math.abs(
            planeY - enemyY
        ) < 50
    ){
        endGame();
        return;
    }

    requestAnimationFrame(
        gameLoop
    );
}
